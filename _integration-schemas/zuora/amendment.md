    replication-key: true
    description: "The date when the amendment was last updated."

  - name: "autoRenew"
    type: "boolean"

  - name: "code"

    type: "string"

      - `Year`
      - `Day`
      - `Week`

  - name: "customerAcceptanceDate"
    type: "date-time"

  - name: "effectiveDate"

  - name: "name"

    type: "string"

      - `RENEW_TO_EVERGREEN`

  - name: "renewalTerm"
    type: "string"

  - name: "renewalTermPeriodType"
    type: "string"
    description: |

      - `Year`
      - `Day`
      - `Week`

    description: "The date when service is activated."

  - name: "status"
    type: "string"
    description: |

      - `Pending Activation`

    type: "string"

    type: "string"

  - name: "type"
    type: "string"
    description: |

      - `NewProduct`
      - `OwnerTransfer`
      - `RemoveProduct`
      - `Renewal`
      - `UpdateProduct`
      - `TermsAndConditions`
      - `SuspendSubscription`
      - `ResumeSubscription`

  - name: "updatedById"
    type: "string"
    description: "The ID of the Zuora user who last updated the amendment."
